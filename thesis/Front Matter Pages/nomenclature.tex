\chapter*{Nomenclature}
\addcontentsline{toc}{chapter}{Nomenclature}

\noindent The following mathematical symbols and notation are used throughout this thesis.

\vspace{0.8cm}

\begin{tabular}{@{}ll@{}}
\textbf{Symbol} & \textbf{Description} \\[0.5em]
$|\psi\rangle$ & Quantum state vector \\[0.3em]
$H$ & Hamiltonian operator \\[0.3em]
$U(\theta)$ & Parameterized unitary gate \\[0.3em]
$\theta$ & Variational parameters \\[0.3em]
$C$ & Cost function \\[0.3em]
$p$ & QAOA circuit depth \\[0.3em]
$n$ & Number of qubits \\[0.3em]
$\gamma, \beta$ & QAOA mixing parameters \\[0.3em]
$\sigma$ & Standard deviation \\[0.3em]
$\mathcal{O}$ & Computational complexity \\[0.3em]
$\otimes$ & Tensor product \\[0.3em]
$\langle \cdot \rangle$ & Expectation value \\[0.3em]
\end{tabular}
